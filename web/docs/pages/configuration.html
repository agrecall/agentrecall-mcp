Configuration & Deployment

Learn how to configure and deploy AgentRecall in various environments, from local development to production.

Environment Variables

AgentRecall uses environment variables for configuration. Create a .env file in the project root:

Required Variables

# Database Configuration
DB_HOST=postgres
DB_PORT=5432
DB_NAME=agentrecall
DB_USER=agentrecall
DB_PASSWORD=your_secure_password_here

# JWT Configuration
JWT_SECRET=your_jwt_secret_at_least_32_bytes_long
JWT_EXPIRES_IN=30d

# OTP Configuration
OTP_MASTER_KEY=your_otp_master_key_at_least_32_bytes_long
OTP_EXPIRES_IN=24h

Optional Variables

# Server Configuration
PORT=3000
NODE_ENV=production
LOG_LEVEL=info

# Redis Configuration
REDIS_HOST=redis
REDIS_PORT=6379
REDIS_PASSWORD=

# Rate Limiting
RATE_LIMIT_WINDOW_MS=900000
RATE_LIMIT_MAX_REQUESTS=100

# CORS
CORS_ORIGIN=https://agentrecall.io

# SSL (Production)
SSL_CERT_PATH=/etc/nginx/ssl/cert.pem
SSL_KEY_PATH=/etc/nginx/ssl/key.pem

Docker Configuration

docker-compose.yml

The main orchestration file for all services:

version: '3.8'

services:
  mcp-server:
    build: ./server
    ports:
      - "3000:3000"
    environment:
      - NODE_ENV=production
    depends_on:
      - postgres
      - redis
    restart: unless-stopped

  postgres:
    image: postgres:15-alpine
    environment:
      POSTGRES_DB: agentrecall
      POSTGRES_USER: agentrecall
      POSTGRES_PASSWORD: ${DB_PASSWORD}
    volumes:
      - postgres_data:/var/lib/postgresql/data
      - ./init.sql:/docker-entrypoint-initdb.d/init.sql
    restart: unless-stopped

  redis:
    image: redis:7-alpine
    volumes:
      - redis_data:/data
    restart: unless-stopped

  nginx:
    image: nginx:alpine
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - ./nginx/nginx.conf:/etc/nginx/nginx.conf
      - ./web/dist:/usr/share/nginx/html
    depends_on:
      - mcp-server
    restart: unless-stopped

volumes:
  postgres_data:
  redis_data:

pgvector Extension

Enable the pgvector extension for vector similarity search:

-- init.sql
CREATE EXTENSION IF NOT EXISTS vector;

-- Create pitfalls table with vector column
CREATE TABLE pitfalls (
    id SERIAL PRIMARY KEY,
    pattern TEXT NOT NULL,
    workaround TEXT NOT NULL,
    embedding vector(1024),
    taxonomy JSONB,
    context_fingerprint TEXT,
    error_signature TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Create vector index for similarity search
CREATE INDEX ON pitfalls USING ivfflat (embedding vector_cosine_ops)
WITH (lists = 100);

Nginx Configuration

Reverse Proxy Setup

# nginx.conf
events {
    worker_connections 1024;
}

http {
    upstream mcp_server {
        server mcp-server:3000;
    }

    server {
        listen 80;
        server_name agentrecall.io;
        return 301 https://$server_name$request_uri;
    }

    server {
        listen 443 ssl http2;
        server_name agentrecall.io;

        ssl_certificate /etc/nginx/ssl/cert.pem;
        ssl_certificate_key /etc/nginx/ssl/key.pem;
        ssl_protocols TLSv1.2 TLSv1.3;
        ssl_ciphers HIGH:!aNULL:!MD5;

        # Static files
        location / {
            root /usr/share/nginx/html;
            try_files $uri $uri/ =404;
        }

        # MCP endpoint
        location /mcp {
            proxy_pass http://mcp_server;
            proxy_http_version 1.1;
            proxy_set_header Upgrade $http_upgrade;
            proxy_set_header Connection 'upgrade';
            proxy_set_header Host $host;
            proxy_cache_bypass $http_upgrade;
        }

        # API endpoint
        location /api {
            proxy_pass http://mcp_server;
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto $scheme;
        }
    }
}

Production Deployment

SSL Certificates

Obtain SSL certificates using Let's Encrypt:

# Install certbot
apt-get update
apt-get install certbot python3-certbot-nginx

# Obtain certificate
certbot --nginx -d agentrecall.io -d www.agentrecall.io

# Auto-renewal
certbot renew --dry-run

Backup Strategy

Implement regular backups for PostgreSQL:

# Backup script
#!/bin/bash
BACKUP_DIR="/backups"
DATE=$(date +%Y%m%d_%H%M%S)
docker exec agentrecall-postgres pg_dump -U agentrecall agentrecall | gzip > $BACKUP_DIR/agentrecall_$DATE.sql.gz

# Keep only last 7 days
find $BACKUP_DIR -name "agentrecall_*.sql.gz" -mtime +7 -delete

Monitoring

Health Checks

Monitor service health:

# Check all services
docker-compose ps

# Check logs
docker-compose logs -f mcp-server

# Health endpoint
curl http://localhost:3000/health
